  | "commods_1d_mom_breadth"
  | "carry_breadth_supported";

export type CurveStateMetricKey = "crude_curve_state" | "natgas_curve_state";

export type RegimeMetricKey = PercentLikeMetricKey | CurveStateMetricKey;

export interface RegimeMetricDescriptor {
  key: RegimeMetricKey;
  label: string;
  kind: "percent" | "state"; // percent shows as 0-100%, state is a qualitative label
}

export interface RegimePanelConfig {
  id: string; // stable identifier, used as the data key
  title: string;
  metrics: RegimeMetricDescriptor[]; // order matters for display
  enabled?: boolean; // future: allow users to toggle
}

export interface RegimePanelData {
  id: EnabledRegimePanelId;
  summary_score?: number; // 0..1 compact read
  metrics: Record<RegimeMetricKey, RegimeMetricValue | undefined>;
  updated_at: string; // ISO timestamp
}

export type RegimeData = Record<EnabledRegimePanelId, RegimePanelData>;

export interface RegimeDataContextValue {
  getPanelData: (id: EnabledRegimePanelId) => RegimePanelData | undefined;
}

export interface RegimeDataProviderProps {
  children: ReactNode;
  update_interval_ms?: number; // default 60s
}

export type RegimeNumeric = number; // 0..1 for percent-like metrics
export type RegimeState = "backwardation" | "contango" | "neutral" | "n/a";
export type RegimeMetricValue = RegimeNumeric | RegimeState;

export interface RegimePanelProps {
  config: RegimePanelConfig;
  onClick?: (id: string) => void;
}

export interface RegimePanelListProps {
  onPanelClick?: (id: string) => void;
}


export type EnabledRegimePanelId = (typeof regime_panels)[number]["id"];
